# ========================= # NUTRISCAN AI - FULL SYSTEM # ========================= import os import io import base64 import requests from datetime import datetime, timedelta from fastapi import FastAPI, Request from dotenv import load_dotenv from sqlalchemy import create_engine, Column, Integer, String, DateTime, Float from sqlalchemy.orm import sessionmaker, declarative_base from PIL import Image from pyzbar.pyzbar import decode from openai import OpenAI # ========================= # ENV CONFIG # ========================= load_dotenv() TELEGRAM_TOKEN = os.getenv("TELEGRAM_BOT_TOKEN") WEBHOOK_URL = os.getenv("WEBHOOK_URL") OPENAI_API_KEY = os.getenv("OPENAI_API_KEY") TELEGRAM_API = f"https://api.telegram.org/bot{TELEGRAM_TOKEN}" client = OpenAI(api_key=OPENAI_API_KEY) # ========================= # DATABASE # ========================= engine = create_engine( os.getenv("DATABASE_URL", "sqlite:///./nutriscan.db"), connect_args={"check_same_thread": False} ) SessionLocal = sessionmaker(bind=engine) Base = declarative_base() class User(Base): __tablename__ = "users" user_id = Column(Integer, primary_key=True) plan = Column(String, default="lite") expiry_date = Column(DateTime, nullable=True) free_trials = Column(Integer, default=3) weight = Column(Float) height = Column(Float) goal = Column(String) class DailyLog(Base): __tablename__ = "daily_logs" id = Column(Integer, primary_key=True) user_id = Column(Integer) date = Column(String) calories = Column(Float, default=0) protein = Column(Float, default=0) carbs = Column(Float, default=0) fats = Column(Float, default=0) Base.metadata.create_all(bind=engine) # ========================= # ETHIOPIAN FOOD DB # ========================= ETHIOPIAN_DB = { "injera": {"calories": 180, "protein": 6, "carbs": 35, "fats": 1}, "doro wat": {"calories": 300, "protein": 25, "carbs": 10, "fats": 20}, "shiro": {"calories": 250, "protein": 15, "carbs": 20, "fats": 12}, "tibs": {"calories": 350, "protein": 30, "carbs": 5, "fats": 25}, } def get_local_food(name): return ETHIOPIAN_DB.get(name.lower()) # ========================= # TELEGRAM HELPERS # ========================= def send_message(chat_id, text): requests.post(f"{TELEGRAM_API}/sendMessage", json={ "chat_id": chat_id, "text": text, "parse_mode": "Markdown" }) def send_photo(chat_id, image_path): requests.post( f"{TELEGRAM_API}/sendPhoto", data={"chat_id": chat_id}, files={"photo": open(image_path, "rb")} ) def get_file(file_id): file = requests.get(f"{TELEGRAM_API}/getFile?file_id={file_id}").json() path = file["result"]["file_path"] return requests.get(f"https://api.telegram.org/file/bot{TELEGRAM_TOKEN}/{path}").content # ========================= # PROGRESS BAR # ========================= def progress_bar(current, total, length=10): if not total: return "░" * length ratio = min(current / total, 1) filled = int(ratio * length) return "█" * filled + "░" * (length - filled) # ========================= # AI ANALYSIS # ========================= def analyze_image(image_bytes): img = base64.b64encode(image_bytes).decode() res = client.chat.completions.create( model="gpt-4o-mini", messages=[{ "role": "user", "content": [ {"type": "text", "text": "Identify Ethiopian food if possible and return JSON: name, calories, protein, carbs, fats"}, {"type": "image_url", "image_url": f"data:image/jpeg;base64,{img}"} ] }] ) return res.choices[0].message.content # ========================= # BARCODE SCAN # ========================= def scan_barcode(image_bytes): image = Image.open(io.BytesIO(image_bytes)) decoded = decode(image) if not decoded: return None code = decoded[0].data.decode("utf-8") res = requests.get(f"https://world.openfoodfacts.org/api/v0/product/{code}.json").json() if res["status"] != 1: return "Product not found." p = res["product"]["nutriments"] return { "calories": p.get("energy-kcal_100g", 0), "protein": p.get("proteins_100g", 0), "carbs": p.get("carbohydrates_100g", 0), "fats": p.get("fat_100g", 0) } # ========================= # TRACKING # ========================= def add_log(db, user_id, data): today = datetime.utcnow().date().isoformat() log = db.query(DailyLog).filter_by(user_id=user_id, date=today).first() if not log: log = DailyLog(user_id=user_id, date=today) db.add(log) log.calories += data["calories"] log.protein += data["protein"] log.carbs += data["carbs"] log.fats += data["fats"] db.commit() return log # ========================= # SUBSCRIPTION # ========================= def check_access(db, user): if user.plan == "lite": if user.free_trials > 0: user.free_trials -= 1 db.commit() return True return False if user.expiry_date and datetime.utcnow() > user.expiry_date: user.plan = "lite" db.commit() return False return True def activate_core(db, user): user.plan = "core" user.expiry_date = datetime.utcnow() + timedelta(days=30) db.commit() # ========================= # PAYMENT MESSAGE # ========================= def payment_message(user_id): return f""" 🔒 Upgrade Required CORE – 199 ETB/month Send to: 0942791319 Use ID: {user_id} Then type: /verify """ # ========================= # FORMAT RESPONSE # ========================= def format_response(data, log): return f""" 🍽 NutriScan Result 🔥 Calories: {data['calories']} kcal 🥩 Protein: {data['protein']}g 🍞 Carbs: {data['carbs']}g 🥑 Fats: {data['fats']}g ━━━━━━━━━━━━━━━ 📊 Daily Progress Calories: {progress_bar(log.calories, 2000)} {int(log.calories)}/2000 Protein: {progress_bar(log.protein, 120)} {int(log.protein)}/120 ━━━━━━━━━━━━━━━ """ # ========================= # FASTAPI APP # ========================= app = FastAPI() @app.post("/webhook") async def webhook(req: Request): data = await req.json() db = SessionLocal() if "message" not in data: return {"ok": True} msg = data["message"] chat_id = msg["chat"]["id"] user_id = msg["from"]["id"] user = db.query(User).filter(User.user_id == user_id).first() if not user: user = User(user_id=user_id) db.add(user) db.commit() if "text" in msg: text = msg["text"] if text == "/start": send_message(chat_id, "Welcome to NutriScan AI 🇪🇹\nSend food image.") return {"ok": True} if text == "/verify": activate_core(db, user) send_message(chat_id, "✅ Subscription Activated") return {"ok": True} if not check_access(db, user): send_message(chat_id, payment_message(user_id)) return {"ok": True} if "photo" in msg: file_id = msg["photo"][-1]["file_id"] image_bytes = get_file(file_id) # Barcode barcode = scan_barcode(image_bytes) if barcode: data = barcode else: ai = analyze_image(image_bytes) # fallback simple parse (improve later) data = { "calories": 400, "protein": 20, "carbs": 50, "fats": 15 } log = add_log(db, user_id, data) send_message(chat_id, format_response(data, log)) return {"ok": True} @app.get("/set-webhook") def set_webhook(): return requests.get(f"{TELEGRAM_API}/setWebhook?url={WEBHOOK_URL}/webhook").json()
